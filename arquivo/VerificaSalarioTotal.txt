function verificaSalarioTotal() {
  // Solicitar a idade do usuário
  let idade;

  do {
    try {
      idade = parseInt(prompt("Digite sua idade:"));
      if (isNaN(idade)) {
        throw new Error("Valor inválido. Digite um número válido.");
      }

      // Verificar se é maior de idade
      if (idade >= 18) {
        console.log("Você é maior de idade.");
      } else {
        console.log("Você é menor de idade.");
      }

    } catch (error) {
      console.log(error.message);
    }
  } while (isNaN(idade));

  // Solicitar o nome do usuário
  const nome = prompt("Digite seu nome:");

  // Solicitar o salário do usuário
  let salario;

  do {
    try {
      salario = parseFloat(prompt("Digite seu salário:"));
      if (isNaN(salario)) {
        throw new Error("Valor inválido. Digite um número válido.");
      }
    } catch (error) {
      console.log(error.message);
    }
  } while (isNaN(salario));

  // Operadores aritméticos
  const bonus = 500;
  const salarioTotal = salario + bonus;
  console.log("Salário com bônus: " + salarioTotal.toFixed(2));
}

// Chamar a função para executar o código
verificaSalarioTotal();
